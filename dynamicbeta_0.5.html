import numpy as np
import matplotlib.pyplot as plt

def SIR_model(S, I, R, beta, gamma, dt, N):
    dS = -(beta * S * I) / N * dt
    dI = ((beta * S * I) / N - gamma * I) * dt
    dR = (gamma * I) * dt

    S += dS
    I += dI
    R += dR

    return S, I, R

# Parameters
N = 21  # Total population
p = 0.025
gamma = 1 / 4
days = 60
dt = 1  # 1 day time step

# Initial conditions
S = 20
I = 1
R = 0

# Store SIR values for each day
S_vals = [S]
I_vals = [I]
R_vals = [R]

# Simulate the SIR model for the specified number of days
for day in range(1, days+1):
    beta = p * S
    S, I, R = SIR_model(S, I, R, beta, gamma, dt, N)
    S_vals.append(S)
    I_vals.append(I)
    R_vals.append(R)

# Plot the results
plt.plot(S_vals, label='Susceptible')
plt.plot(I_vals, label='Infected')
plt.plot(R_vals, label='Recovered')
plt.axhline(y=0, color='gray', linestyle='--', label='Infected=0') # add horizontal line at y=0
plt.xlabel('Day')
plt.xlabel('Day')
plt.ylabel('Number of Kids')
plt.xticks(np.arange(0, days+1, 5))
plt.title(f'SIR Model for {days} Days')

# Set custom y-axis tick labels
plt.yticks(np.arange(0, 21, 1))
plt.legend()
plt.show()

def simulate_tommy_infections(n_simulations, susceptible_kids, p):
    infections = []
    for _ in range(n_simulations):
        infected_kids = np.random.binomial(susceptible_kids, p)
        infections.append(infected_kids)
    return infections

def simulate_day2_infections(n_simulations, susceptible_kids, p):
    total_infections = []
    for _ in range(n_simulations):
        # Tommy infects kids on Day 1
        infected_by_tommy = np.random.binomial(susceptible_kids, p)
        
        # Remaining susceptible kids after Day 1
        remaining_susceptible_kids = susceptible_kids - infected_by_tommy
        
        # Kids infected by Tommy on Day 1 might infect others on Day 2
        infected_on_day2 = np.random.binomial(remaining_susceptible_kids, p)
        
        # Total infected kids by Day 2 (including Tommy)
        total_infected = 1 + infected_by_tommy + infected_on_day2
        total_infections.append(total_infected)
    return total_infections

# Parameters
n_simulations = 1000
susceptible_kids = 20
p = 0.02

# Run simulations
day2_infections = simulate_day2_infections(n_simulations, susceptible_kids, p)

# Calculate the expected number of kids infected by Day 2
expected_infected_kids_day2 = np.mean(day2_infections)
print(f'Expected number of kids infected by Day 2: {expected_infected_kids_day2:.2f}')
